  \chantindent

  \chantindent

    {Tan-tang-lai, tee-tan dai-tuk, kor-hai-tan mee-kwam-suk, tan-tang-lai, tee-tan-dai-suk, kor-hai-suk-ying-ying-keurn-bai.}
