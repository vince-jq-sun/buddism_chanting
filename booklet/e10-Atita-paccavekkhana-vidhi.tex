\section*{e10 | Atīta-paccavekkhaṇa-vidhi}

\begin{chantsubsection}
  \chantnote
    {\{Handa mayaṃ atītapaccavekkhaṇapāṭhaṃ bhaṇāmase.\}}
  \chantnotetrans
    {\{Now let us recite the reflection on past use.\}}

\chantgap
  % Cīvara (robe)
  \chantlineinline
    {Ajja mayā apaccavekkhitvā yaṃ cīvaraṃ paribhuttaṃ,}
    {Today, without proper reflection, I used a robe;}
  \chantlineinline
    {taṃ yāvadeva sītassa paṭighātāya,}
    {only to ward off cold,}
  \chantlineinline
    {uṇhassa paṭighātāya,}
    {to ward off heat,}
  \chantlineinline
    {ḍaṃsamakasavātātapasiriṃsapasamphassānaṃ paṭighātāya,}
    {to ward off contact with flies, mosquitoes, wind, sun, and creeping things,}
  \chantlineinline
    {yāvadeva hirikopinapaṭicchādanatthaṃ.}
    {and only for the purpose of covering the private parts.}

\chantgap

  % Piṇḍapāta (almsfood)
  \chantlineinline
    {Ajja mayā apaccavekkhitvā yo piṇḍapāto paribhutto,}
    {Today, without proper reflection, I consumed almsfood;}
  \chantlineinline
    {so neva davāya na madāya na maṇḍanāya na vibhūsanāya,}
    {not for amusement, not for intoxication, not for adornment, not for beautification,}
  \chantlineinline
    {yāvadeva imassa kāyassa ṭhitiyā yāpanāya vihiṃsuparatiyā brahmacariyānuggahāya,}
    {but only for the maintenance and nourishment of this body, for ending harm, and for supporting the holy life;}
  \chantlineinline
    {iti purāṇañca vedanaṃ paṭihaṅkhāmi navañca vedanaṃ na uppādessāmi,}
    {thus I shall allay old feelings and not arouse new feelings,}
  \chantlineinline
    {yātrā ca me bhavissati anavajjatā ca phāsuvihāro cāti.}
    {and I shall live blamelessly and at ease.}

\chantgap

  % Senāsana (lodging)
  \chantlineinline
    {Ajja mayā apaccavekkhitvā yaṃ senāsanaṃ paribhuttaṃ,}
    {Today, without proper reflection, I used lodging;}
  \chantlineinline
    {taṃ yāvadeva sītassa paṭighātāya,}
    {only to ward off cold,}
  \chantlineinline
    {uṇhassa paṭighātāya,}
    {to ward off heat,}
  \chantlineinline
    {ḍaṃsamakasavātātapasiriṃsapasamphassānaṃ paṭighātāya,}
    {to ward off contact with flies, mosquitoes, wind, sun, and creeping things,}
  \chantlineinline
    {yāvadeva utuparissayavinodanaṃ paṭisallānārāmatthaṃ.}
    {only for the removal of discomforts due to the seasons and for the delight of seclusion.}

\chantgap

  % Bhesajja (medicinal requisites)
  \chantlineinline
    {Ajja mayā apaccavekkhitvā yo gilānapaccayabhesajjaparikkhāro paribhutto,}
    {Today, without proper reflection, I used medicines and requisites for the sick;}
  \chantlineinline
    {so yāvadeva uppannānaṃ veyyābādhikānaṃ vedanānaṃ paṭighātāya,}
    {only to ward off arisen painful feelings,}
  \chantlineinline
    {abyāpajjhaparamatāyāti.}
    {and for the utmost freedom from affliction.}
\end{chantsubsection}
